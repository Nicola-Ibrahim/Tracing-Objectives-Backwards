\tikzset{
    axis/.style={->, >=stealth, line width=0.8pt},
    diag/.style={dashed, gray, line width=0.6pt},
    pitbar/.style={fill=blue!30, draw=blue!70},
    cdfline/.style={line width=1.4pt},
    regime/.style={font=\tiny\bfseries, opacity=0.6}
}

% ================= 1. OVERCONFIDENT (TOP-LEFT) =================
\begin{scope}[shift={(0,0)}]
    \node[anchor=south] at (2,4.2) {\textbf{Overconfident} (U-shaped)};
    \draw[axis] (0,0) -- (4,0) node[right, font=\tiny] {PIT};
    \draw[axis] (0,0) -- (0,1.5);
    \draw[pitbar] (0.2,0) rectangle (0.8,1.2);
    \draw[pitbar] (1.1,0) rectangle (1.7,0.4);
    \draw[pitbar] (2.3,0) rectangle (2.9,0.4);
    \draw[pitbar] (3.2,0) rectangle (3.8,1.2);
    
    \begin{scope}[shift={(1,1.8)}]
        % Shading regimes
        \fill[red!10] (0,0) -- (0,2) -- (2,2) -- cycle;
        \node[regime, red, anchor=north west] at (0,1.8) {Overconfident};
        
        \draw[diag] (0,0) -- (2,2);
        \draw[axis] (0,0) -- (2.2,0);
        \draw[axis] (0,0) -- (0,2.2);
        \draw[cdfline, blue] (0,0) .. controls (0, 1.8) and (2, 0.2) .. (2,2);
    \end{scope}
    \node[text width=4.5cm, align=center, font=\scriptsize] at (2,-0.8) {Variance too low; truth falls in tails.};
\end{scope}

% ================= 2. UNDERCONFIDENT (TOP-RIGHT) =================
\begin{scope}[shift={(6.5,0)}]
    \node[anchor=south] at (2,4.2) {\textbf{Underconfident} (Humped)};
    \draw[axis] (0,0) -- (4,0) node[right, font=\tiny] {PIT};
    \draw[axis] (0,0) -- (0,1.5);
    \draw[pitbar] (0.2,0) rectangle (0.8,0.3);
    \draw[pitbar] (1.1,0) rectangle (2.9,1.3);
    \draw[pitbar] (3.2,0) rectangle (3.8,0.3);
    
    \begin{scope}[shift={(1,1.8)}]
        % Shading regimes
        \fill[orange!10] (0,0) -- (2,0) -- (2,2) -- cycle;
        \node[regime, orange, anchor=south east] at (2,0.2) {Underconfident};
        
        \draw[diag] (0,0) -- (2,2);
        \draw[axis] (0,0) -- (2.2,0);
        \draw[axis] (0,0) -- (0,2.2);
        \draw[cdfline, orange!80!black] (0,0) .. controls (1.8, 0.1) and (0.2, 1.9) .. (2,2);
    \end{scope}
    \node[text width=4.5cm, align=center, font=\scriptsize] at (2,-0.8) {Variance too high; truth falls in center.};
\end{scope}

% ================= 3. BIASED / SKEWED (BOTTOM-LEFT) =================
\begin{scope}[shift={(0,-6.5)}]
    \node[anchor=south] at (2,4.2) {\textbf{Biased / Skewed}};
    \draw[axis] (0,0) -- (4,0) node[right, font=\tiny] {PIT};
    \draw[axis] (0,0) -- (0,1.5);
    \draw[pitbar] (0.2,0) rectangle (1.2,1.3);
    \draw[pitbar] (1.5,0) rectangle (2.5,0.6);
    \draw[pitbar] (2.8,0) rectangle (3.8,0.2);
    
    \begin{scope}[shift={(1,1.8)}]
        % Bias usually stays in one regime
        \fill[red!5] (0,0) -- (2,0) -- (2,2) -- cycle;
        \node[regime, red, anchor=south east] at (2,0.2) {Under-predicting};
        
        \draw[diag] (0,0) -- (2,2);
        \draw[axis] (0,0) -- (2.2,0);
        \draw[axis] (0,0) -- (0,2.2);

    \end{scope}

\end{scope}

% ================= 4. WELL-CALIBRATED (BOTTOM-RIGHT) =================
\begin{scope}[shift={(6.5,-6.5)}]
    \node[anchor=south] at (2,4.2) {\textbf{Well-Calibrated} (Uniform)};

    \begin{scope}[shift={(1,1.8)}]

        \draw[axis] (0,0) -- (2.2,0);
        \draw[axis] (0,0) -- (0,2.2);
        \draw[cdfline, green!60!black] (0,0) -- (2,2);
        \node[regime, green!60!black] at (1,1.3) {Perfect};
    \end{scope}
    \node[text width=4.5cm, align=center, font=\scriptsize] at (2,-0.8) {Sampler matches true distribution.};
