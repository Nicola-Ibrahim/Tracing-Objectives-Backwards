\begin{tikzpicture}[
    % --- Global Configuration via tikzset ---
    node distance=2.2cm and 2.8cm, % Slightly increased for better breathing room
    font=\sffamily,
    >=Stealth,
]

\tikzset{
    % Node Base Styles
    base/.style={
        rectangle, rounded corners=8pt, align=center,
        line width=1pt, % Slightly thicker for a bolder, modern look
        drop shadow={opacity=0.15, shadow xshift=1.5pt, shadow yshift=-1.5pt}
    },
    % Specific Component Styles
    goalNode/.style={
        base, draw=blue!50!black, top color=white, bottom color=blue!10, 
        text=black, minimum width=2.8cm, minimum height=1.4cm, font=\sffamily\small
    },
    engineNode/.style={
        base, draw=orange!60!black, top color=white, bottom color=orange!10, 
        text=black, rounded corners=14pt, minimum width=3.6cm, minimum height=1.8cm, font=\sffamily\small
    },
    envNode/.style={
        draw=gray!60!black, ellipse, top color=white, bottom color=gray!15, text=black,
        minimum width=3.6cm, minimum height=1.8cm, align=center, font=\sffamily\small, line width=1pt,
        drop shadow={opacity=0.15, shadow xshift=1.5pt, shadow yshift=-1.5pt}
    },
    candidateNode/.style={
        base, draw=green!50!black, top color=white, bottom color=green!10, 
        text=black, minimum width=2.8cm, minimum height=1.4cm, font=\sffamily\small
    },
    % Connection Styles
    connector/.style={
        ->, color=gray!70, line width=1.2pt, shorten >=3pt, shorten <=3pt
    },
    loopLabel/.style={
        font=\sffamily\tiny\itshape, text=black!80, align=center,
        fill=white!90!orange!10, % Subtle background so dashed lines don't ruin text
        inner sep=2pt, rounded corners=3pt
    },
    callout/.style={
        font=\sffamily\tiny, text=#1, inner sep=2pt
    }
}

% ================= 1. Placement of Main Nodes =================

% Input/Goals
\node[goalNode] (goals) {Design Space\\Constraints \& Goals\\(Sparse Labels)};

% Explorer Engine (Central)
\node[engineNode, right=of goals] (engine) {Exploration Engine\\(RL Agent, MCTS,\\or Heuristic PSO)};

% Environment/Simulator (Below)
\node[envNode, below=2.0cm of engine] (simulator) {Environment /\\Physics Simulator};

% Output/Candidates
\node[candidateNode, right=of engine] (output) {Diverse High-Scoring\\Candidate Designs};

% ================= 2. Annotations & Background =================

% Drawn before connections so the loop box stays neatly in the back
\begin{scope}[on background layer]
    % Increased inner ysep to explicitly stretch the height of the block
    \node[fill=orange!4, rounded corners=24pt, draw=orange!35, 
          line width=1.2pt, dashed, fit=(engine) (simulator), 
          inner xsep=25pt, inner ysep=45pt] (loop_bg) {};
    
    \node[anchor=south, font=\sffamily\scriptsize\bfseries, text=orange!80!black, yshift=6pt] 
          at (loop_bg.south) {Active Exploration Loop};
\end{scope}

% ================= 3. Connections & Loops =================

% Trigger Flow
\draw[connector] (goals) -- (engine);

% Active Exploration Loop
\draw[connector, bend left=45] (engine.east) to 
    node[right, loopLabel, xshift=2pt] {Action\\(Design Selection)} (simulator.east);

\draw[connector, bend left=45] (simulator.west) to 
    node[left, loopLabel, xshift=-2pt] {Observation / Reward\\(Property)} (engine.west);

% Discovery Path
\draw[connector] (engine) -- node[above, loopLabel, fill=white, yshift=2pt] {Discovery} (output);

% ================= 4. Method Sub-descriptions & Callouts =================

\node[below=0.1cm of engine, callout=gray!80!black, text width=3cm] 
    {Iteratively optimizes\\sequences of decisions};

\node[below=0.1cm of simulator, callout=gray!80!black, text width=3.5cm] 
    {Provides feedback where\\gradients may be unavailable};

% Strength Callouts
\node[above=0.2cm of engine, callout=blue!70!black] {High Exploration Efficiency};
\node[above=0.2cm of output, callout=green!60!black] {Population Diversity};

\end{tikzpicture}
