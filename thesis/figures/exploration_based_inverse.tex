\begin{tikzpicture}[
    % --- Global Configuration via tikzset ---
    node distance=2.0cm and 2.5cm,
    font=\sffamily,
    >=Stealth,
]

\tikzset{
    % Node Base Styles
    base/.style={
        rectangle, rounded corners=6pt, align=center,
        line width=0.8pt, drop shadow={opacity=0.1, shadow xshift=1pt, shadow yshift=-1pt}
    },
    % Specific Component Styles (No Bold)
    goalNode/.style={
        base, draw=blue!40, top color=white, bottom color=blue!5, 
        text=blue!70!black, minimum width=2.6cm, minimum height=1.3cm, font=\sffamily\small
    },
    engineNode/.style={
        base, draw=orange!50, top color=white, bottom color=orange!5, 
        text=orange!70!black, rounded corners=12pt, minimum width=3.4cm, minimum height=1.6cm, font=\sffamily\small
    },
    envNode/.style={
        draw=black!40, ellipse, fill=gray!5, text=black!70,
        minimum width=3.4cm, minimum height=1.6cm, align=center, font=\sffamily\small, line width=0.8pt,
        drop shadow={opacity=0.05, shadow xshift=1pt, shadow yshift=-1pt}
    },
    candidateNode/.style={
        base, draw=green!40!black!50, top color=white, bottom color=green!5, 
        text=green!50!black, minimum width=2.6cm, minimum height=1.3cm, font=\sffamily\small
    },
    % Connection Styles
    connector/.style={
        ->, color=gray!60, line width=1.1pt, shorten >=2pt, shorten <=2pt
    },
    loopLabel/.style={
        font=\sffamily\tiny\itshape, text=gray!80, align=center
    },
    callout/.style={
        font=\sffamily\tiny, text=#1, inner sep=2pt
    }
}

% ================= 1. Placement of Main Nodes =================

% Input/Goals
\node[goalNode] (goals) {Design Space\\Constraints \& Goals\\(Sparse Labels)};

% Explorer Engine (Central)
\node[engineNode, right=of goals] (engine) {Exploration Engine\\(RL Agent, MCTS,\\or Heuristic PSO)};

% Environment/Simulator (Below)
\node[envNode, below=1.8cm of engine] (simulator) {Environment /\\Physics Simulator};

% Output/Candidates
\node[candidateNode, right=of engine] (output) {Diverse High-Scoring\\Candidate Designs};

% ================= 2. Connections & Loops =================

% Trigger Flow
\draw[connector] (goals) -- (engine);

% Active Exploration Loop
\draw[connector, bend left=45] (engine.east) to 
    node[right, loopLabel, xshift=3pt] {Action\\(Design Selection)} (simulator.east);

\draw[connector, bend left=45] (simulator.west) to 
    node[left, loopLabel, xshift=-3pt] {Observation / Reward\\(Property)} (engine.west);

% Discovery Path
\draw[connector] (engine) -- node[above, loopLabel, yshift=2pt] {Discovery} (output);

% ================= 3. Annotations & Background =================

% Background Highlight for the Loop
\begin{scope}[on background layer]
    \node[fill=orange!2, rounded corners=18pt, draw=orange!15, 
          line width=1pt, densely dashed, fit=(engine) (simulator), inner sep=20pt] (loop_bg) {};
    
    \node[anchor=south, callout=orange!80, yshift=2pt] 
          at (loop_bg.south) {Active Exploration Loop};
\end{scope}

% Method Sub-descriptions
\node[below=0.1cm of engine, callout=gray!80, text width=3cm]

    {Provides feedback where\\gradients may be unavailable};

% Strength Callouts

\node[above=0.2cm of output, callout=green!50!black] {Population Diversity};
