    inverse/.style={base, draw=colInv, fill=colInv!10, text=colInv, font=\sffamily\bfseries\footnotesize, minimum width=3.4cm},

    target/.style={base, fill=white, draw=black, minimum width=2.6cm, double, double distance=1pt},
    ranking/.style={base, draw=colSelect, fill=colSelect!10, text=colSelect!80!black, minimum width=3.4cm},
    check/.style={base, shape=diamond, aspect=1.8, inner sep=0pt, draw=colFwd, fill=colFwd!5, text=colFwd!80!black, minimum width=3.6cm},
    result/.style={base, fill=colSuccess, draw=colSuccess!70!black, text=white, font=\sffamily\bfseries\footnotesize, minimum width=3.4cm},

    % =============================================================================

    % =============================================================================
    \begin{scope}[local bounding box=OfflineScope]
        \node[titleStyle] (Title1) at (0, 0.8) {MODEL SYNTHESIS};

        \node[data, below=0.5cm of Title1] (RawData) {Raw Examples $\mathcal{D}_{raw}$};
        \node[norm, below=of RawData] (Preprocessing) {Normalization \&\\Preprocessing};
        \node[inverse, below=of Preprocessing] (InverseModel) {Inverse Model\\$q_\theta(x|y)$};

        \draw[->, colTrain] (RawData) -- (Preprocessing);
        \draw[->, colTrain] (Preprocessing) -- node[right, font=\tiny] {Fit Generator} (InverseModel);

        \begin{scope}[on background layer]
            \node[draw=black!10, fill=black!2, dashed, rounded corners=12pt, inner sep=15pt, fit=(Title1) (InverseModel)] (Box1) {};

        \node[target, below=0.5cm of Title2] (Target) {Target Outcome\\$y^\star \in \mathcal{Y}$};
        \node[inverse, below=of Target] (Propose) {Proposal Mechanism\\Generate $\{\hat{x}^{(k)}\}_{k=1}^K$};

        \node[forward, right=1cm of Propose] (Surrogate) {Forward Surrogate\\$\hat{f}(x)$};

        % Ranking receiving insights
        \node[ranking, below=0.8cm of Check] (Ranking) {Selection \& Ranking\\Score by Discrepancy};
        
        \node[result, below=0.8cm of Ranking] (Decision) {Validated Decision $\hat{x}$};

        % Internal Connections
        \draw[->] (Target) -- (Propose);
        \draw[->] (Propose) -- node[right, font=\tiny] {Candidate Cloud} (Check);
        \draw[->, colFwd] (Surrogate) |- (Check);
        \draw[->] (Check) -- node[right, font=\tiny] {Insights} (Ranking);
        \draw[->] (Ranking) -- node[right, font=\scriptsize\bfseries, colSuccess] {Success} (Decision);

        % Online Container
        \begin{scope}[on background layer]
            \node[draw=black!10, fill=black!2, dashed, rounded corners=12pt, inner sep=15pt, fit=(Title2) (Decision) (Surrogate)] (Box2) {};
